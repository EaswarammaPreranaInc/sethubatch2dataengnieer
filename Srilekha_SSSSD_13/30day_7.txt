n = int(input("enter any positive number:"))

for i in range(n):
    for j in range(n - i - 1):
        print(' ', end='')
    for k in range(2 * i + 1):
        print('*', end='')
    print()

n = int(input("enter a number:")) 

for i in range(n):
    if i == 0:
        print(' ' * (n - i - 2) + '*')  
    else:
        print(' ' * (n - i - 1) + '*' * (2 * i + 1))
