intersection()   method  demo  program (Home  work)
a = {10 , 20 , 30 , 40}                                    # set
b = {30 , 40 , 50 , 60}                                    # set
c = a . intersection(b)                                    # {30,40}
print(c)                                                   # {30,40}  common elements in both set
print(type(c))                                             # set
d = a & b                                                  # {30,40}
print(d)                                                   # {30,40}
print(type(d))                                             # set
print(c  is  d)                                            #  false because different objects
print(c  ==  d)                                            # True values are same



 difference()   method  demo  program  (Home  work)
a = {10 , 20 , 30 , 40}                                       # set
b = {30 , 40 , 50 , 60}                                       # set
c = a . difference(b)                                         # {10,20}
print(c)                                                      #{10,20} the set which is in a but not in b
print(type(c))                                                # set
d = a - b                                                     # difference objects
print(d)                                                      #{10,20}
print(type(d))                                                #set
print(c  is  d)                                               # False, because  c and d are different objects
print(c  ==  d)                                               #true because values are same

#  in  and  not  in  operators  (Home  work)

print(30  in  a . keys())                                                                    # True
print(60  in  a . keys())                                                                     # Flase
print(60  in  a . values())                                                                  # True
print(30  in  a . values())                                                                  # False
print(50  in  a)                                                                             # True
print(20  in  a)                                                                             # False
print(70  not  in  a . keys())                                                               # False

print(25  not  in  a)                                                                        # True
